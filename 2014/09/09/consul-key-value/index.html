階層化とdecode

BSD の base64 コマンドだと
-dがdebugで-Dがdecodeだった

['-']% curl -X PUT -d 'bazbazbaz' -s 'http://127.0.0.1:8500/v1/kv/foo/bar/baz'
true
['x']%
['x']% curl -s 'http://127.0.0.1:8500/v1/kv/foo/bar/baz' | jq '.[]'
{
  "CreateIndex": 565,
  "ModifyIndex": 568,
  "LockIndex": 0,
  "Key": "foo/bar/baz",
  "Flags": 0,
  "Value": "YmF6YmF6YmF6"
}
['-']%
['-']%
['-']% curl -s 'http://127.0.0.1:8500/v1/kv/foo/bar/baz' | jq '.[].Value' -r | base64 -D
bazbazbaz
['-']%
